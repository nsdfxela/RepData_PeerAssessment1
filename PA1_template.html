isWeekend <- function(day){
  if(day=="Saturday"||day=="Sunday")
    TRUE
  else FALSE
}
dayType <- sapply(weekdays(datadfImp$trueTime), isWeekend)
datadfImp <- cbind(datadfImp, dayType)

Separating data tables by dayType:

dataWEImp <- datadfImp[dayType==TRUE,]
